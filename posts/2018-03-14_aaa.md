```go
func main() {
  ch := make(chan int);

  wg := &sync.WaitGroup{}
  wg.Add(1)

  go func() {
    defer wg.Done()

    ch <- "OK"
  }()

  wg.Wait()

  close(ch)

  for v := range ch {
    fmt.Println(ch)
  }
}
```
